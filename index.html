📘 Class Blueprint: WeatherStation
class WeatherStation:
    def __init__(self, name, elev, temp):
        self.name = name
        self.elevation = elev
        self.temp = temp

    def calibrate_temp(self):
        self.temp = self.temp - (self.elevation * 0.006)
